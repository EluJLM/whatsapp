  match /databases/{database}/documents {
    match /{document=**} {
      // Permitir a todos los usuarios leer
      allow read: if true;
      
      // Permitir escribir solo a usuarios espec√≠ficos
      allow write: if request.auth != null && request.auth.uid in [
        "UID_USUARIO_1",
        "UID_USUARIO_2",
        "UID_USUARIO_3"
      ];
    }
  }
}
